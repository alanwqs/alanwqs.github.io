# 704.二分查找

二分查找

给定一个 n 个元素有序的（升序）整型数组 nums 和一个目标值 target ，写一个函数搜索 nums 中的 target，如果目标值存在返回下标，否则返回 -1。

输入: nums = [-1,0,3,5,9,12], target = 9  
输出: 4       
解释: 9 出现在 nums 中并且下标为 4

提示：

  • 你可以假设 nums 中的所有元素是不重复的。
  • n 将在 [1, 10000]之间。
  • nums 的每个元素都将在 [-9999, 9999]之间。
var search = function(nums, target) {
    // 定义左区间
    let left = 0;
    // 定义右区间
    let right = nums.length - 1;

    // 当在区间作闭右闭时，=依然有效
    while(left <= right){
        // 相当于 （left + right） / 2，防止溢出
        let middle = Math.floor(left + ((right - left) / 2))
        if(nums[middle] < target) {
            // 当target大于中间值时，说明在右边区间
            left = middle + 1
        } else if(nums[middle] > target){
            // 当target小于中间值时，说明在左边区间
            right = middle - 1
        } else {
            // 说明正好是中间
            return middle
        }
    }
    // 没有找到
    return -1
};

# 35.搜索插入位置

给定一个排序数组和一个目标值，在数组中找到目标值，并返回其索引。如果目标值不存在于数组中，返回它将会被按顺序插入的位置。

请必须使用时间复杂度为 O(log n) 的算法。

var searchInsert = function(nums, target) {
    let left = 0;
    let right = nums.length - 1

    while(left<=right){
        let middle = Math.floor(left + (right - left) /2)
        if(nums[middle] === target) {
            return middle
        } else if(nums[middle] < target) {
            left = middle + 1
        } else if(nums[middle] > target) {
            right = middle - 1
        }
    }
    // 分别处理如下四种情况
    // 目标值在数组所有元素之前  [0, -1]
    // 目标值等于数组中某一个元素  return middle;
    // 目标值插入数组中的位置 [left, right]，return  right + 1
    // 目标值在数组所有元素之后的情况 [left, right]， return right + 1
    return right + 1 
};

# 34.在排序数组中查找元素的第一个和最后一个位置

给定一个按照升序排列的整数数组 nums，和一个目标值 target。找出给定目标值在数组中的开始位置和结束位置。

如果数组中不存在目标值 target，返回 [-1, -1]。

输入：nums = [5,7,7,8,8,10], target = 8
输出：[3,4]
1
2
var searchRange = function(nums, target) {
    // 找到左边的第一个元素，返回right
    function binarySearcgLeft(nums,target) {
        let left = -1
        let right = nums.length;
        while(left + 1 !== right){
            let mid = Math.floor((left + right) / 2)
            if(nums[mid] >= target) {
                right = mid
            }else{
                left = mid
            }
        }
        return right
    }
    // 找到右边的最后一个元素，返回left
    function binarySearcgRight(nums,target) {
        let left = -1
        let right = nums.length;
        while(left + 1 !== right){
            let mid = Math.floor((left + right) / 2)
            if(nums[mid] > target) {
                right = mid
            }else{
                left = mid
            }
        }
        return left
    }
    // 获取两个索引
    let leftIndex = binarySearcgLeft(nums,target)
    let rightIndex = binarySearcgRight(nums,target)
    // 判断左边索引<=右边索引 && 右边索引不能越界 && 左右索引===target
    if(leftIndex<=rightIndex && leftIndex < nums.length && nums[leftIndex] === target && nums[rightIndex]===target){
        return [leftIndex,rightIndex]
    }else{
        return [-1,-1]
    }
};

# 69.x的平方根

给你一个非负整数 x ，计算并返回 x 的 平方根 。

由于返回类型是整数，结果只保留 整数部分 ，小数部分将被 舍去 。

注意：不允许使用任何内置指数函数和算符，例如 pow(x, 0.5) 或者 x ** 0.5 。

var mySqrt = function(x) {
    if(x<2) return x
    let left = 1;
    let right = Math.floor(x/2)
    while(left <= right){
        let mid = Math.floor(left + (right-left)/2)
        if(mid*mid===x){
            return mid
        }else if(mid*mid>x){
            right = mid -1
        }else{
            left = mid+1
        }
    }
    return right
};

# 26.删除有序数组重复项

给你一个有序数组 nums ，请你 原地 删除重复出现的元素，使每个元素 只出现一次 ，返回删除后数组的新长度。

不要使用额外的数组空间，你必须在 原地 修改输入数组 并在使用 O(1) 额外空间的条件下完成。

// 暴力法
var removeElement = function(nums, val) {
    let len = nums.length
    for(let i = 0;i<len;i++){
        // 当发现相等元素时，后面元素往前移动一步
        if(nums[i] === val){
            for(let j = i+1;j<len;j++){
                nums[j-1] = nums[j]
            }
            // 因为下表i以后的数值都向前移动了一位，所以i也向前移动一位
            i--;
            // 此时数组的大小-1
            len--
        }
    }
    return len
};

注意：如果对空文本输入退格字符，文本继续为空。

TIP

输入：S = "ab#c", T = "ad#c"

输出：true

解释：S 和 T 都会变成 “ac”。

// 模拟栈的方法
// 时间复杂度:O(n + m)

var backspaceCompare = function(s, t) {
    let myS = ''

    for(let i = 0; i < s.length; i++){

            myS += s[i]
        } else {
            myS = myS.slice(0,-1)
        }

        if(t[i]!=='#'){
            myT += t[i]
        } else {
            myT = myT.slice(0,-1)
        }
    }
    if(myS !== myT){
        return false
    }else{
        return true
    }
};

# 977. 有序数组的平方

给你一个按 非递减顺序 排序的整数数组 nums，返回 每个数字的平方 组成的新数组，要求也按 非递减顺序 排序。

TIP

输入：nums = [-4,-1,0,3,10]

输出：[0,1,9,16,100]

解释：平方后，数组变为 [16,1,0,9,100]

排序后，数组变为 [0,1,9,16,100]

const sortedSquares = function (nums) {
    let res = []
    for (let i = 0, j = nums.length - 1; i <= j;) {
      const left = Math.abs(nums[i])
      const right = Math.abs(nums[j])
      if (right > left) {
        // push element to the front of the array
        res.unshift(right * right)
        j--
      } else {
        res.unshift(left * left)
        i++
      }
    }
    return res
  }

# 209. 长度最小的子数组

给定一个含有 n 个正整数的数组和一个正整数 target 。

找出该数组中满足其和 ≥ target 的长度最小的 连续子数组 [numsl, numsl+1, ..., numsr-1, numsr] ，并返回其长度。如果不存在符合条件的子数组，返回 0 。

TIP

输入：target = 7, nums = [2,3,1,2,4,3]

输出：2

解释：子数组 [4,3] 是该条件下的长度最小的子数组。

var minSubArrayLen = function(target, nums) {
    let len = nums.length
    let left = 0
    let right = 0
    let sum = 0
    let res = len + 1
    while(right < len){
        sum += nums[right]
        right++
        while(sum >= target){
            res = res < right - left ? res : right - left
            sum -= nums[left]
            left++
        }
    }
    return res > len ? 0 : res
};

# 76. 最小覆盖子串

给你一个字符串 s 、一个字符串 t 。返回 s 中涵盖 t 所有字符的最小子串。如果 s 中不存在涵盖 t 所有字符的子串，则返回空字符串 "" 。

注意：

  • 对于 t 中重复字符，我们寻找的子字符串中该字符数量必须不少于 t 中该字符数量。
  • 如果 s 中存在这样的子串，我们保证它是唯一的答案。

TIP

输入：s = "ADOBECODEBANC", t = "ABC"

输出："BANC"

var minWindow = function(s, t) {
    let left = 0;
    let right = 0;

    let res = '';

    // 存储需要的字母，数量
    const map = new Map()
    for(let i=0;i<t.length;i++){
        map.set(t[i], map.get(t[i])?map.get(t[i])+1:1)
    }

    // 需要的类型有几种
    let needType = map.size

    while(right<s.length){
        // 获取当前遍历的字符
        const cur = s[right]
        // 如果map里面有，那么我们可以从当前位置开始计算
        if(map.has(cur)){
            // 数量减 1
            map.set(cur,map.get(cur)-1)
            // 如果刚好只需要一个，那么类型也减 1
            if(map.get(cur)===0){
                needType-=1
            }
        }
        // 说明已经符合条件
        while(needType === 0){
            const leftChar = s[left]
            // 获取当前符合条件的字符串
            let newRes = s.slice(left,right+1)
            // 更新字符串
            if(!res||res.length>newRes.length){
                res = newRes
            }

            // 左指针移动过程中，要是有一个字符脱离了，那就加上重新滑动窗口
            if(map.has(leftChar)){
                map.set(leftChar,map.get(leftChar)+1)
                if(map.get(leftChar)===1){
                    needType += 1
                }
            }
            // 移动左指针
            left++
        }
        // 移动右指针
        right++
    }
    return res
};

            return slow;

var repeatedSubstringPattern = function (s) {
    if (s.length === 0)
        return false;

    const getNext = (s) => {
        let next = [];
        let j = 0;

        next.push(j);

        for (let i = 1; i < s.length; ++i) {
            while (j > 0 && s[i] !== s[j])
                j = next[j - 1];
            if (s[i] === s[j])
                j++;
            next.push(j);
        }

        return next;
    }

    let next = getNext(s);
    // 如果 next[len - 1] != -1，则说明字符串有最长相同的前后缀（就是字符串里的前缀子串和后缀子串相同的最长长度）

var maxSlidingWindow = function(nums, k) {
    let queue = []
    let ans = []
    for(let i = 0;i<nums.length;i++){
        // 若队列不为空，且当前元素大于等于队尾所存下标的元素，则弹出队尾
        while(queue.length && nums[i] >= nums[queue[queue.length - 1]]){
            queue.pop()
        }
        // 入队当前元素下标
        queue.push(i)
        // 判断当前最大值（即队首元素）是否在窗口中，若不在便将其出队
        while(queue[0] <= i - k){
            queue.shift()
        }
        // 当达到窗口大小时便开始向结果中添加数据
        if(i >= k - 1){
            ans.push(nums[queue[0]])
        }
    }
    return ans
};

# 理论基础

  • 满二叉树：

如果一棵二叉树只有度为0的结点和度为2的结点，并且度为0的结点在同一层上，则这棵二叉树为满二叉树。

  • 完全二叉树

在完全二叉树中，除了最底层节点可能没填满外，其余每层节点数都达到最大值，并且最下面一层的节点都集中在该层最左边的若干位置。若最底层为第 h 层，则该层包含 1~ 2^h -1 个节点。

  • 二叉搜索树

前面介绍的树，都没有数值的，而二叉搜索树是有数值的了，二叉搜索树是一个有序树。

若它的左子树不空，则左子树上所有结点的值均小于它的根结点的值；

若它的右子树不空，则右子树上所有结点的值均大于它的根结点的值；

它的左、右子树也分别为二叉排序树

  • 平衡二叉搜索树

又被称为AVL（Adelson-Velsky and Landis）树，且具有以下性质：它是一棵空树或它的左右两个子树的高度差的绝对值不超过1，并且左右两个子树都是一棵平衡二叉树。
