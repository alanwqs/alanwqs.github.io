                            # 2. 两数相加

                            思路：

                            • 确定变量

                            需要定义临时的相加节点sum，新链表的头指针head，移动的新链表sum

                            • 确定循环操作

                            while(flag || l1 || l2)表示，最后一位有进位，链表1或2还有值的时候可以进入循环

                            • 指针移动

                            三者的指针都需要同时移动

                            //   function ListNode(val, next) {
//     this.val = (val===undefined ? 0 : val)
//     this.next = (next===undefined ? null : next)
//   }
var addTwoNumbers = function(l1, l2) {
    // 定义是否相加超过 10
    let flag = 0;
    // 定义相加的节点
    let sum = new ListNode(0);
    // 最后返回的指针
    let head = sum;
    // 当有进位 / l1还有 / l2还有
    while(flag || l1 || l2){
        let v1 = l1.val ? l1.val : 0
        let v2 = l2.val ? l2.val : 0
        let res = v1+v2+flag
        // 更新进位
        flag = res >= 10 ? 1: 0
        // 下一个节点
        sum.next = new ListNode(res%10)
        // 移动新链表
        sum = sum.next
        // 移动两个链表
        if(l1){
            l1 = l1.next
        }
        if(l2){
            l2 = l2.next
        }
    }
    return head.next
};

                            # 4. 寻找两个正序数组的中位数

                            暴力解法的时间复杂度是O(NlogN)，也就是排序的复杂度，我们可以用另一种方法进行优化

                            // 暴力解法
var findMedianSortedArrays = function(nums1, nums2) {
    let n = nums1.length + nums2.length;

    // 合并数组并重新排序
    let num = nums1.concat(nums2).sort((a,b)=>a-b)

    // 长度是偶数就取中间两个/2，奇数就直接取中间的
    return n % 2 === 0 ? (num[n/2] + num[n/2-1])/2 : num[Math.floor(n/2)]
};

                            题目表明了两个数组是有序的，并且长度我们是知道的，我们只要找到两个数组的中位数索引，便能找到中位数。

                            我们可以使用双指针的方式，分别指向两个数组，从左到右进行遍历。记录当前的值，并且移动两个指针更新

                            此时的 时间复杂度O(n+m)，n为nums1的长度，m为nums2的长度

                            var findMedianSortedArrays = function(nums1, nums2) {
    // 计算数组总长度
    let len = nums1.length + nums2.length;

    // 记录前一个值
    let pre = -1;
    // 记录当前值
    let cur = -1;

    // num1上面的指针
    let p1 = 0;
    // num2上面的指针
    let p2 = 0;

    // 循环到数组的一半
    for(let i = 0;i<=Math.floor(len/2);i++){
        // 更新前一个值
        pre = cur;
        // 当p1没到终点，并且p2到达终点 或 p1指向的值小于p2指向的值
        if(p1 < nums1.length && (p2 >= nums2.length || nums1[p1] < nums2[p2])){
            cur = nums1[p1];
            p1++;
        } else {
            cur = nums2[p2];
            p2++;
        }
    }

    return len % 2 === 0 ? (cur + pre) / 2 : cur
};

                            # 5. 最长回文子串

                            给你一个字符串 s，找到 s 中最长的回文子串。

                            TIP

                            输入：s = "babad"

                            输出："bab"

                            解释："aba" 同样是符合题意的答案。

                            var longestPalindrome = function(s) {
    if(s.length < 2) return s;
    let res = ''
    for(let i = 0;i<s.length;i++){
        helper(i,i)
        helper(i,i+1)
    }

    function helper(start,end){
        while(start >=0 && end < s.length && s[start] === s[end]){
            start--;
            end++;
        }
        if(end - start - 1 > res.length){
            res = s.slice(start + 1, end)
        }
    }
    return res
};

                            # 7. 整数反转

                            TIP

                            输入：x = 123

                            输出：321

                            var reverse = function(x) {
    let sign = Math.sign(x)
    x = Math.abs(x)

    x = x.toString().split('').reverse().join('')
    
    let res = x*sign

    let max = Math.pow(2,31);
    let min = -Math.pow(2,31)

    if(res>min && res<max){
        return res
    }
    return 0
};

                            # 11. 盛最多水的容器

                            var maxArea = function(height) {
    let left = 0;
    let right = height.length-1;

    let res = 0

    while(left<right){
        if(height[left] > height[right]){
            res = Math.max(res,height[right] * (right-left))
            right--;
        } else {
            res = Math.max(res,height[left] * (right-left))
            left++;
        }
    }

    return res
};

                            # 14. 最长公共前缀

                            TIP

                            输入：strs = ["flower","flow","flight"]

                            输出："fl"

                            var longestCommonPrefix = function(strs) {
    let s = strs[0]
    if(!s) return ''

    let res = ''

    for(let i = 0;i<s.length;i++){
        let flag = strs.every((item)=>item[i]===s[i])
        if(flag) res += s[i]
        else return res
    }
    return res
};

                            # 15. 三数之和

                            给你一个包含 n 个整数的数组 nums，判断 nums 中是否存在三个元素 a，b，c ，使得 a + b + c = 0 ？请你找出所有和为 0 且不重复的三元组。

                            注意：答案中不可以包含重复的三元组。

                            var threeSum = function(nums) {
    if(!nums) return [];
    if(nums.length<3) return []
    let ans = []
    nums.sort((a,b)=>a-b)

    for(let i = 0;i<nums.length;i++){
        if(nums[0]>0) break;

        if(i>0 && nums[i]===nums[i-1]) continue;

        let L = i + 1;
        let R = nums.length - 1;

        while(L < R){
            const sum = nums[i] + nums[L] + nums[R]
            if(sum === 0){
                ans.push([nums[i],nums[L],nums[R]])
                while(L<R && nums[L]===nums[L+1]){
                    L++;
                }
                while(L<R && nums[R]===nums[R-1]){
                    R--;
                }
                L++;
                R--;
            } else if(sum>0){
                R--;
            } else if(sum<0){
                L++;
            }
        }
    }
    return ans;
};

                            # 16. 最接近的三数之和

                            给定一个包括 n 个整数的数组 nums 和 一个目标值 target。找出 nums 中的三个整数，使得它们的和与 target 最接近。返回这三个数的和。假定每组输入只存在唯一答案。

                            TIP

                            输入：nums = [-1,2,1,-4], target = 1

                            输出：2

                            解释：与 target 最接近的和是 2 (-1 + 2 + 1 = 2) 。

                            var threeSumClosest = function(nums, target) {
    let len = nums.length;
    nums.sort((a,b)=>a-b)
    let res = nums[0]+nums[1]+nums[2]

    for(let i = 0;i<len;i++){
        let L = i+1;
        let R = len-1;
        while(L<R){
            const sum = nums[i] + nums[L] + nums[R]
            if(Math.abs(target - sum) < Math.abs(target - res)){
                res = sum
            } else if(sum > target){
                R--;
            } else if(sum < target){
                L++;
            } else {
                return res
            }
        }
    }
    return res
};

                            # 20. 有效的括号

                            给定一个只包括 '('，')'，'{'，'}'，'['，']' 的字符串 s ，判断字符串是否有效。

                            var isValid = function(s) {
    let stack = []
    let map = new Map()
    map.set('{','}')
    map.set('[',']')
    map.set('(',')')

    for(let i = 0;i<s.length;i++){
        if(map.has(s[i])){
            stack.push(map.get(s[i]))
        }else {
            if(stack.pop()!==s[i]){
                return false
            }
        }
    }

    return !stack.length
};

                            # 21. 合并两个有序链表

                            var mergeTwoLists = function(l1, l2) {
    if(l1===null) return l2;
    if(l2===null) return l1;

    if(l1.val < l2.val){
        l1.next = mergeTwoLists(l1.next,l2)
        return l1
    } else {
        l2.next = mergeTwoLists(l1,l2.next)
        return l2
    }
};

                            # 23. 合并K个升序链表

                            var mergeKLists = function(lists) {
    let list = []
    for(let i = 0;i<lists.length;i++){
        let item = lists[i]
        while(item){
            list.push(item.val)
            item = item.next
        }
    }
    list.sort((a,b)=>a-b)
    let head = new ListNode()
    let now = head
    for(let i =0;i<list.length;i++){
        let node = new ListNode(list[i])
        now.next = node
        now = now.next
    }

    return head.next
};

                            # 26. 删除有序数组中的重复项

                            var removeDuplicates = function(nums) {
    let slow = 0
    for(let fast = 1;fast<nums.length;fast++){
        if(nums[slow]!==nums[fast]){
            slow++;
            nums[slow] = nums[fast]
        }
    }
    return slow + 1
};
